var todo = {
  finished: true,
  tasks: [
    { task: 'Read book', done: true },
    { task: 'Walk', done: false }
  ],
  getFinished() {
    return this.tasks.filter(function(item){
      return item.done === this.finished;
    });
  }
};
console.log(todo.getFinished())
