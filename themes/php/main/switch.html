Конструкция switch-case


switch ($i) {
    case 0: echo "i равно 0"; break;
    case 1: echo "i равно 1"; break;
    default: echo "Значение по умолчанию"; break;
}



$car = "Ferrari";
switch($car) {
	case "Ferrari";
		$year = 2012;
		echo "$year";
		break;
	case "BMW";
		$year = 2012;
		echo "$year";
		break;
}
