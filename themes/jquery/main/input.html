<input type="checkbox"> (галочка)

var checkbox = $('#checkbox').prop('checked'); // читаем (true/false) $('#checkbox').prop('checked', true); // устанавливаем значение checked (true/false)

<input type="radio"> (переключатель)

var radio = $('input[name=radio]:checked').prop('id'); // получаем id выбранного radio var radio = $('#radio1').prop('checked'); // проверяем по id выбран ли radio (true/false) $('#radio1').prop('checked', true); // устанавливаем checked по id

<select></select> (выпадающий список)

var value = $('#select').val(); // читаем выбранный value var text = $('#select option:selected').text(); // читаем выбранный text var text = $('#select [value=2]').text(); // читаем text по заданному value $('#select [value=3]').attr('selected', 'selected'); // устанавливаем заданный элемент в selected
