Выборка элементов страницы


$('p');     // по названию тега
$('.box');  // по названию класса
$('#box');  // по идентификатору ID
$('div a'); // вложенные теги

Атрибуты


$('img[width=200]');      // по точному значению атрибута
$('input[name=message]'); // по атрибуту (имени)
$('a[href^=http]');       // по началу значению атрибута (^ - начинается)
$('img[scr$=.jpg]');      // по окончанию значению атрибута ($ - заканчивается)
$('img[scr*=moto]');      // по вхождению в значение атрибута (* - в любом месте)

Фильтры


$('#table tr:even');       // фильтр четности - четные элементы, нумерация с "0" (0, 2 и т.д.)
$('#table tr:odd');        // фильтр нечетности - нечетные элементы
$('img:not(#box img)');    // фильтр отрицания. Все кроме...
$('div:has(fieldset)');    // фильтр элементов, которые содержат другие элементы (has - имеет)
$('p:contains(мотоцикл)'); // фильтр по содержанию определенного текста (contains - содержит)
$('#box img:first');       // фильтр первого элемента
$('#box img:last');        // фильтр последнего элемента
$('div:hidden');           // фильтр для всех скрытых блоков
$('div:visibility');       // фильтр для всех видимых блоков
