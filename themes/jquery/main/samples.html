Примеры

Сообщения, возвращаемые от php скриптов


function echo(text) {
    $('#echo').html(text);
    $('#echo').slideDown(700);
    setTimeout(function() { $('#echo').slideUp(700) }, 2700);
}

Поиск по товарам


$('#search').keyup(function() {             // search - input[text]
    var value = $('#search').val();         // получаем вводимые символы
    value = value.toLowerCase();            // переводим в нижний регистр
    $('.container').css('display', 'none'); // container - содержит блоки с названиями товаров
    $('.container:contains("'+value+'")').css('display', 'block'); // показываем блоки
});

Закрытие формы


$(document).ready(function(){
    var formOut = true;
    $('#box').mouseleave(function(){ formOut = true; });
    $('#box').mouseenter(function(){ formOut = false; });
    $('.bg').click( function() { if (formOut) { $('.bg').fadeOut(500); } });
});
