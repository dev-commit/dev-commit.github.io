События hover() и toggle()

hover()


// событие hover() - наведение мыши (объединяет в себе mouseover и mouseout)
$('#box').hover(function() {
    // наведение мыши
},function() {
    // увод мыши
});

toggle()


// переключатель (2 положения)
$('#formHide').toggle(function() {
    $('#form').fadeOut(1000);
	$(this).text('Развернуть форму');
}, function() {
    $('#form').fadeIn(1000);
	$(this).text('Свернуть форму');
});
