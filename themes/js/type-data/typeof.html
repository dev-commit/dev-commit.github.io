typeof

typeof - возвращает тип аргумента


typeof 0            // "number"
typeof 'foo'        // "string"
typeof true         // "boolean"
typeof undefined    // "undefined"
typeof null         // "object" (*)
typeof {}           // "object"
typeof []           // "object"
typeof new Date     // "object"
typeof function(){} // "function" (*)


// проверка поддержки свойств
typeof addEventListener !== 'undefined'
