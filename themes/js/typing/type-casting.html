Приведение типов


// && - первое значение
// || - последнее значение

false && ''         // false
false || ''         // ''
'' || 'Yes'         // 'Yes'
{} && 'Some'        // 'Some'
{} || 'Some'        // Object {}
{a: 10} && 'Some'   // 'Some'
{a: 10} || 'Some'   // Object { a: 10}
0 || true           // true
null || 0           // 0
undefined && 0      // undefined
'' || 0 && true     // 0
{} || 0 && true     // Object {}
false || {} && true // true
