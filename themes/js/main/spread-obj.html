Spread-оператор объектов

  • Работает как Object.assign()

Пример

const defaults = { host: 'localhost', user: 'admin' }; // более высокий приоритет const options = { user: 'Tony', password: 'qwerty' } // user из options перезапишет user из defaults // => { host: 'localhost', user: 'Tony', password: 'qwerty' } // комбинация свойств const port = 8080; const result = { ...defaults, ...options, port, connect() { // } };
