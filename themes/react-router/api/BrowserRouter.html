BrowserRouter

  • BrowserRouter - router. Использует html5 history api и следит за тем чтобы UI был синхронизирован с тем что написано в адресной строке
  • BrowserRouter технически является компонентом, но не создаёт DOM-элемента
import { BrowserRouter, Switch, Route, NavLink } from 'react-router-dom'; const App = () => ( <> <BrowserRouter basename="/root/"> <NavLink to="/" exact>Home</NavLink> <Switch> <Route path='/' exact component=HomePage /> </Switch> </BrowserRouter> </> )
