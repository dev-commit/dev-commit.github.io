<h1>Hello</h1> <button onclick="alert('Hello')">Кнопка</button> <input type="text" /> /// <reference types="cypress" /> describe('Starter', () => { beforeEach(() => { cy.visit('http://localhost:8081/index'); // открыть url для теста cy.pause(); // остановить выполнение и продолжить вручную }); it('Starter', () => { cy.contains('Hello'); // на странице содержится 'Hello' cy.get('button').click(); // вызов события cy.url().should('include' 'index'); // проверка url на вхождение 'index' cy.checkTitle('Hello'); // вызов собственного метода из './cypress/support/commands.js' }); it('Check value', () => { // изменить и проверить value cy.get('input').type('New Value').should('have.value', 'New Value'); }); it('Without Cypress', () => { expect(2).to.equal(2); }); }); // Реализация своего метода Cypress.Commands.add('checkTitle', (message) => { cy.visit('http://localhost:8081/index'); cy.contains(message); }) cy .get('element') .children() .should('contain', 'Some text') .should('not.contain') .and('be.visible')
