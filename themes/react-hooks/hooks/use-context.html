Хук контекста useContext

Можно подписаться на контекст React без использования каких-либо вложений объект контекста (значение, возвращённое из React.createContext) Возвращает текущее значение контекста для принятого контекста
  • Текущее значение контекста определяется пропом value ближайшего <MyContext.Provider> над вызывающим компонентом в дереве
import React, { useContext } from 'react'; const value = useContext(MyContext); const App = () => { const locale = useContext(LocaleContext); const theme = useContext(ThemeContext); // ... }

Примеры

Пример #1. Установка языка

import React from 'react'; import ReactDOM from "react-dom"; import LanguageContextProvider from './LanguageContextProvider'; import { useLanguage } from './LanguageContextProvider'; const App = () => ( <LanguageContextProvider> <Component /> </LanguageContextProvider> ); const Component = (props) => { const isLanguage = useLanguage(); return ( <> Hello {isLanguage} </> ); }; const rootElement = document.getElementById("root"); ReactDOM.render(<App />, rootElement); import React, { useContext } from 'react'; const IsLanguageContext = React.createContext('ru'); const LanguageContextProvider = ({children, ...props}) => ( <IsLanguageContext.Provider value={'en'} {...props}> {children} </IsLanguageContext.Provider> ); export default LanguageContextProvider; export const useLanguage = () => useContext(IsLanguageContext);
