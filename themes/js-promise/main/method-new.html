new Promise

Promise

Объявление
let promise = new Promise((resolve, reject) => {
    if (...) {
    	// onResolved()
        resolve('ok');
    } else {
    	// onRejected()
        reject(new Error('Api error')); // аналогично с throw
        throw new Error('Api error');   // аналогично с reject
    }
});

Использование
promise
    .then(result => {
        console.log(result);
    })
    .catch(error => {
        console.error(error);
        console.error(error.message); 
    })

Объявление
function asyncFunc() {
	let promise = new Promise((resolve, reject) => {
		//
	});
	return promise;
}

Использование
asyncFunc()
    .then(result => { ... })
    .catch(result => { ... })
