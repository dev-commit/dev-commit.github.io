Примеры

getByText

import React from 'react'; import App from './App'; import { render } from '@testing-library/react'; it('App', () => { const { getByText } = render(<App />); // вхождение слова "Hello" в текст компонента expect(getByText(/hello/i)).toBeInTheDocument(); }); import React from 'react'; const App = () => (<div>Hello</div>); export default App;

getByTestId

import React from 'react'; import App from './App'; import { render, screen, cleanup } from '@testing-library/react'; // для рендера одного компонента несколько раз afterEach(cleanup); it('App', () => { const { getByTestId } = render(<App lable="My Label" />); expect(screen.getByTestId('button')).toBeInTheDocument(); }); import React from 'react'; const App = ({ label }) => ( <div data-testid="button"> {label} </div> ); export default App;

Create Snapshot

import React from 'react'; import { render, cleanup } from '@testing-library/react'; import App from './App'; it('App', () => { const { asFragment } = render(<App label="My Label" />); expect(asFragment()).toMatchSnapshot(); }); import React from 'react'; const App = ({ label }) => ( <header> <h1>{label}</h1> </header> ); export default App;
  • После запуска теста, создается файл ./__snapshots__/App.test.js.snap
