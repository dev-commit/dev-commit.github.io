simulate

Имитация событий на корневом узле в "wrapper". Должен быть один узел Имя события для симуляции Mock-объект события, который будет передан обработчикам событий
import React, { Component } from 'react'; import { shallow } from 'enzyme'; class App extends Component { constructor(props) { super(props); this.state = { name: 'Tony', job: '' }; } handleClick = () => { this.setState({ name: 'Jack' }); } handleChange = ({ target: { value } }) => { this.setState({ job: value }); } render() { return ( <div> <button onClick={this.handleClick}>Ok</button> <input onChange={this.handleChange} type="text"/> </div> ); } } it('', () => { const wrapper = shallow(<App />); // 1. Начальное состояние name: 'Jack'. expect(wrapper.state('name')).toBe('Tony'); // 2. Изменили состояние: name: 'Tony'. wrapper.find('button').simulate('click'); // 3. Проверили корректность изменения состояния. expect(wrapper.state('name')).toBe('Jack'); }); it('', () => { const wrapper = shallow(<App />); // 1. Начальное состояние job: ''. expect(wrapper.state('job')).toBe(''); // 2. Изменили состояние: job: 'Developer'. wrapper.find('input').simulate('change', { target: { value: 'Developer' } }); // 3. Проверили корректность изменения состояния. expect(wrapper.state('job')).toBe('Developer'); });
