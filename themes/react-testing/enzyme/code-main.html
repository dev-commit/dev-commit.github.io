import React from 'react'; import { shallow } from 'enzyme'; const Child = ({ propChild }) => ( <div>{propChild}</div> ); const App = ({ propApp }) => { return ( <div> <h1 className="header">Hello</h1> <Child propChild="Hello Child" /> <h1>Hello Last</h1> <a target="_blank"></a> </div> ) } it('', () => { const wrapper = shallow(<App propApp="Hello App" />); expect(!!wrapper) .toBeTruthy(); expect(wrapper.find('h1').first().text()) .toBe('Hello'); expect(wrapper.find('h1').last().text()) .toBe('Hello'); expect(wrapper.find('h1').at(0).text()) .toBe('Hello'); expect(wrapper.find('h1').text()) .toBe('Hello'); expect(wrapper.find('h1').text()) .toContain('ello'); expect(wrapper.find('h1').html()) .toBe('<h1 class="header">Hello</h1>'); expect(wrapper.contains(<h1 className="header">Hello</h1>)) .toBeTruthy(); expect(wrapper.find('h1').length) .toBe(1); expect(wrapper.find('h2').length) .toBe(0); expect(wrapper.find('h1').exists()) .toBeTruthy(); expect(wrapper.find('h1').hasClass('header')) .toBeTruthy(); expect(wrapper.find('h1').hasClass('nothing')) .toBeFalsy(); // Аналогично что для React-компонента. expect(wrapper.find(Child).props().propChild) .toBe('Hello Child'); expect(wrapper.find(Child).prop('propChild')) .toBe('Hello Child'); // Аналогично что для DOM-элемента. expect(wrapper.find('a').props().target) .toBe('_blank'); expect(wrapper.find('a').prop('target')) .toBe('_blank'); expect(wrapper.find(Child).text('propChild')) .toBeTruthy(); }); expect(wrapper.find('h1').length).toBe(1); // есть элемент expect(wrapper.find('h2').length).toBe(0); // нет элемента
