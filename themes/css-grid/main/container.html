Grid Container


display: grid; # объявление


grid-template-columns: 300px 300px 300px;                     # столбцы (3 штуки)
grid-template-rows:    150px 150px 150px;                     # строки (3 штуки), можно не задавать. адаптируются под высоту контента
grid-template:         150px 150px 150px / 300px 300px 300px; # короткая запись строки / столбцы
grid-auto-columns:     200px;                                 # размер по умолчанию для неявно заданных столбцов
grid-auto-rows:        200px;                                 # размер по умолчанию для неявно заданных строк


grid-gap: 20px;         # расстояние между строками и стобцами (толщина линии)
grid-column-gap: 20px;  # расстояние между столбцами
grid-row-gap: 20px;     # расстояние между строками


grid-auto-flow: row;    # порядок элементов по очереди по строкам (по умолчанию)
grid-auto-flow: column; # порядок элементов по очереди по столбцам
  • Дляgrid-template-columns и grid-template-rows

300px 300px 300px;                     # 3 элемента (фиксированные величины)
1fr 1fr 1fr;                           # 3 элемента (относительные величины) fr - фракция
500px 1fr 2fr;                         # 3 элемента (смешанные величины)
500px auto 500px;                      # 3 элемента (auto - занимает ширину контента)
repeat(3, 1fr);                        # задать повторяющийся паттерн
minmax(150px, 250px) 1fr 1fr;          # 3 элемента, первый с minmax
minmax(100px, auto);                   # определяется автоматически по ширине элементов
repeat(auto-fill, 200px);              # добавлять столько элементов фиксированых размеров, сколько поместится в контейнер
repeat(auto-fill, minmax(150px, 1fr)); # когда минимальное значение 150px перескакивают колонки (может остаться пустое пространство сбоку)
repeat(auto-fit, minmax(150px, 1fr));  # когда минимальное значение 150px перескакивают колонки (пустое пространство не оставляет, блок заполняет всю ширину)


repeat()    # задание повторяющегося паттерна
minmax()    # задание минимального и максимального значения
auto-fill() # растяжение дочерних grid элементов на всю ширину родителя, если хватает элементов
auto-fit()  # растяжение дочерних grid элементов на всю ширину родителя
  • fr можно применять к высоте, только если у родителя задана высота
