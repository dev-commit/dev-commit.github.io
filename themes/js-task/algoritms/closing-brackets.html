Проверить закрывающиеся скобки

  • Напишите функцию тестирующую правильность вложенных скобок трех типов(кроме нижеперчисленных символов других в строке нет) - '(' и ')', '[' и ']', '<' и '>':
test('[]()<>') // => true test('[]()<)') // => false test('[(<>)]') // => true test('[(<>])') // => false function test (str) { var strArr = str.split(''); var stack = []; var check = { ']': '[', ')': '(', '>': '<' }; strArr.forEach(function (bracket) { if ((stack.length === 0) || !check[bracket] ) { stack.push(bracket); } else { var lastElmInStack = stack[stack.length - 1]; if (lastElmInStack === check[bracket]) { stack.pop(); } } }); return !stack.length; }
