Пересечение массивов

const arr1 = [1, 2, 3, 4, 5]; const arr2 = [4, 5, 6, 7, 8]; const arr3 = []; const obj = {} arr2.forEach(el => { obj[el] = true; }) arr1.forEach(element => { if (obj[element]) { arr3.push(element); } }) console.log(arr3); // => [4, 5]
