Палиндром

  • Палиндром - число, буквосочетание, слово или текст, одинаково читающееся в обоих направлениях
function isPalindrom(str) { const length = Math.round(str.length / 2); for(let i = 0; i < length; i++) { if (str[i] !== str[str.length - i - 1]) { return false; } } return true; } console.log(isPalindrom('ротор')); // true console.log(isPalindrom('ротор1')); // false
