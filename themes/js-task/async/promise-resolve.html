    .then((val) => {
        console.log(val);
        throw new Error();
    })
    .then((val) => console.log('then', val))
    .catch((val) => {
        console.log('catch', val);
        return Promise.reject();
    })
    .then(finalHandler, finalHandler1)
    .then(finalHandler, finalHandler1)
    .then(finalHandler, finalHandler1);

    console.log('finalHandler')
}

    console.log('finalHandler1')
}
