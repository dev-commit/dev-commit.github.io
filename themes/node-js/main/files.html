Файлы и директории

Чтение из файла

Синхронно. Дальнейший код будет выполняться только после чтения файла (Sync) var fs = require('fs'); var file_readed = fs.readFileSync('text.txt', 'utf8'); console.log(file_readed); Асинхронно var fs = require('fs'); fs.readFile('text.txt', 'utf8', function(err, data) { console.log(data); });

Запись в файл

Синхронно var fs = require('fs'); fs.writeFileSync('new_file.txt', 'Текст для \n файла'); Асинхронно var fs = require('fs'); fs.writeFile('new_file.txt', 'Текст для \n файла', function(err, data) {});

Удаление файла

var fs = require('fs'); fs.unlinkSync('file.txt'); // синхронно fs.unlink('file.txt', function() {}); // асинхронно // удаление сначала файла из папки fs.unlink('./test-folder/file.txt', function() { // после этого удаление папки fs.rmdir('test-folder', function() {}); });
