Циклы For, While


@for $i from through - диапазон включает значение начала и конца
@for $i from to      - диапазон включает значение начала и не включает значение конца



@for $i from 1 through 3 {
	.item-#{$i} {
		width: 2em * $i;
		background-image: url(img/#{$i}.jpg);
	}
}









.item-1 {
	width: 2em;
	background-image: url(img/1.jpg);
}
.item-2 {
	width: 4em;
	background-image: url(img/2.jpg);
}
.item-3 {
	width: 6em;
	background-image: url(img/3.jpg);
}



$i: 7;
@while $i > 0 {
	.item-#{$i} {
		width: 1em * $i;
	}
	$i: $i - 3;
}





.item-7 {
	width: 7em;
}
.item-4 {
	width: 4em;
}
.item-1 {
	width: 1em;
}
