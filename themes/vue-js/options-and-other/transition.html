Анимация


<transition name="fade"></transition>
<transition name="slide-down"></transition>

Примеры

#1 - Плавное появление


.fade-enter-active, .fade-leave-active {
	transition: opacity 0.4s;
}
.fade-enter, .fade-leave-to {
	opacity: 0;
}

#2 - Всплытие снизу


.element {
	position: fixed;
    bottom: 25px;
    left: 25px;
}
.fade-enter-active, .fade-leave-active {
    transition: 1s;
}
.fade-enter, .fade-leave-to {
    transform: translateY(100px);
}

#3 - Keyframes


.fade-enter-active {
	animation: fade 0.4s;
}
.fade-leave-active {
	animation: fade 0.4s reverse;
}
@keyframes fade {
	0%   { transform: scale(0); }
	50%  { transform: scale(1.5); }
	100% { transform: scale(1); }
}
