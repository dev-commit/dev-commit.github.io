Передача событий между компонентами

  • Создание событий. Глобальный метод $emit()

Прослушиване событий


<app-child @close="modal = false"></app-child>   <!-- родительский компонент -->
<button @click="$emit('close')"></button>        <!-- дочерний компонент -->


<app-child @close="modal = $event"></app-child>  <!-- родительский компонент -->
<button @click="$emit('close', true)"></button>  <!-- дочерний компонент -->

Пример


<div id="app">
	<ul>
		<li v-for="message in messages">{{ message }}</li>
	</ul>
	<message @message-saved="handleMessage"></message>
</div>


Vue.component('message', {
	template: '<input type="text" v-model="message" @keyup.enter="saveMessage">',
	data: function() {
		return {
			message: ''
		}
	},
	methods: {
		saveMessage() {
			this.$emit('message-saved', this.message); // передача события родителю
			this.message = '';
		}
	}
})

var app = new Vue({
	el: '#app',
	data: {
		messages: []
	}
	methods: {
		handleMessage(message){
			this.messages.push(message)
		}
	}
})
