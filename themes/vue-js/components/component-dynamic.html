Динамические компоненты

Создание компонента


<component :is="currentView"></component>

Кэширование компонента (сохранение состояния)


<keep-alive>
	<component :is="currentView"></component>
</keep-alive>

Пример


<!-- если нужна анимация, component обернуть в transition -->
<div id="app">
	<nav>
		<a href="#" @click="changeView('orange')">orange</a>
		<a href="#" @click="changeView('blue')">blue</a>
	</nav>
	<component :is="currentView"></component>
</div>

<template id="orange-template">
	<div class="orange-block"></div>
</template>
<template id="blue-template">
	<div class="blue-block"></div>
</template>



var app = new Vue({
	el: '#app',
	data: {
		currentView: 'orange'
	},
	components: {
		orange: {
			template: '#orange-template'
		},
		blue: {
			template: '#blue-template'
		}
	},
	methods: {
		changeView(view) {
			this.currentView = view
		}
	}
})

},
beforeCreate() {
	this.$options.components.TreeFolderContents = require('./icons/icon-shopping-cart.vue').default
}

require code 2


<template>
	<component :is="$options.components[name]"></component>
</template>


<script>
export default {
	props: {
		name: { type: String }
	},
	created() {
		this.$options.components[this.name] = require('./icons/icon-'+this.name+'.vue').default;
	}
}
</script>

Подшрузка компонента с данными из роутинга


<template>
    <app-theme />
</template>

<script>
export default {
    components: {
        'app-theme': null,
    },
    created() {
        this.$options.components['app-theme'] = 
            require(`../category/${this.$route.params.name}.vue`).default;
    }
}
</script>
