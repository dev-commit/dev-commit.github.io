Геттеры (getters)

Установка геттера в Vuex.Store


const store = new Vuex.Store({
    state: {
        count: 0
    },
    getters: {
        storeCount(state) {
            return state.count
        }
    }
})

Чтение геттера


<template>
    <div class="container">Корзина покупок {{ count }}</div>
</template>


<script>
export default {
    computed: {
        count() {
            return this.$store.getters.storeCount
        }
    }
}
</script>

Чтение геттера (mapGetters)


<template>
    <div class="container">Корзина покупок {{ storeCount }}</div>
</template>


<script>
import { mapGetters } from 'vuex'

export default {
    computed: {
    	// смешиваем результат mapGetters с внешним объектом computed
        ...mapGetters([
            'storeCount'
        ])
    }
}
</script>


// указание другого имени
export default {
    computed: {
        ...mapGetters({
            count: 'storeCount'
        })
    }
}
