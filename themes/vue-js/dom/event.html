Доступ к оригинальному событию DOM

  • $event - доступ к оригинальному событию DOM, event нативное событие DOM
  • event.target - выбранный элемент
  • this внутри методов указывает на экземпляр Vue

<button @click="methodOne"></button>
<button @click="methodTwo($event, id)"></button>


var app = new Vue({
	el: '#app',
	methods: {
		methodOne(event) {
			if (event) {
				console.log(event.target.tagName);
				event.preventDefault();
			}
		},
		methodTwo(event, id) {
			event.toElement.classList.add('active');
		}
	}
})
