Объект события (Event)

  • Event - объект созданный перед запуском события
  • Когда происходит событие, браузер создаёт объект события, записывает в него детали и передаёт его в качестве аргумента функции-обработчику

const btn = document.getElementById('btn');
btn.addEventListener('click', getButton);

function getButton(event) {
	const a = event.type;             // название события (н-р click)
	const a = event.target;           // инициатор события
	const a = event.currentTarget;    // элемент на котором сработал обработчик. Значение – обычно такое же, как и у this, но если обработчик является функцией-стрелкой или при помощи bind привязан другой объект в качестве this, то мы можем получить элемент из event.currentTarget
	const a = event.eventPhase;       // номер фазы, на которой событие было поймано (Event.CAPTURING_PHASE=1, Event.AT_TARGET=2, Event.BUBBLING_PAGE=3)
	const a = event.timeStamp;        // объект Date, указывающий, когда произошло событие
	const a = event.bubbles;          // true если событие может всплывать по дереву элементов. события blur, focus, load, unload не всплывают
	const a = event.cancelable;       // true если с этим событием связано действие по умолчанию и его можно отменить preventDefault
	const a = event.clientX;          // координата X курсора в момент клика относительно окна, для событий мыши
	const a = event.clientY;          // координата Y курсора в момент клика относительно окна, для событий мыши

	event.preventDefault();           // отключение действия события по умолчанию
	event.stopPropagation();          // прекращение всплытия
	event.stopImmediatePropagation(); // прекращение всплытия и остановка обработки событий на текущем элементе
}
