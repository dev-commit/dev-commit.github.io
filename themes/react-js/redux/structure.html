redux/reducers/my-reducer.js
import { ADD_VALUE } from '../actionTypes';

const initialState = ['Default Value 1', 'Default Value 2'];

const myReducer = (state = initialState, action) => {
    switch (action.type) {
        case ADD_VALUE: {
            return [...state, action.payload.myValue];
        }
        default: {
            return state;
        }
    }
};

export default myReducer;

4. ACTIONS (ACTIONS CREATOR)

redux/actions.js
import { ADD_VALUE } from './actionTypes';

export const addMyValue = myValue => ({
    type: ADD_VALUE,
    payload: { myValue }
})

5. COMPONENT

App.js
import React, {Component} from 'react';
import { connect } from 'react-redux';
import { addMyValue } from './redux/actions';

class App extends Component {
	componentDidMount() {
		this.props.addMyValue('New Value');
	}
	render() {
		console.log(this.props.myValue);
		// => (2) ["Default Value 1", "Default Value 2"]
		// => (3) ["Default Value 1", "Default Value 2", "New Value"]
		return (<div></div>);
	}
}

const mapStateToProps = state => ({
	myValue: state.myReducer
})
const mapDispatchToProps = {
	addMyValue: addMyValue
}

export default connect(
	mapStateToProps,
	mapDispatchToProps
)(App);
