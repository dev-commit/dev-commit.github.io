makeStyles and withStyles

makeStyles


import { makeStyles } from '@material-ui/core/styles';
const useStyles = makeStyles(theme => ({
	label: {
		mardin: '10px',
	},
}));
const MyComponent = () => {
	const classes = useStyles();
	return (<label className={classes.label} />);
}

withStyles

MyComponent.js
import { withStyles } from '@material-ui/core/styles';
import styles from './MyComponentStyles';
const MyComponent = () => {
	return (<label className={classes.label} />);
};
export default withStyles(styles)(CRMRadio);

MyComponentStyles.js
const MyComponentStyles = () => ({
    label: {
		mardin: '10px',
    }
});
export default MyComponentStyles;
