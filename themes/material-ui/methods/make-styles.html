makeStyles

Простой пример

import { makeStyles } from '@material-ui/core/styles'; const useStyles = makeStyles(theme => ({ label: { mardin: '10px', }, })); const Component = () => { const classes = useStyles(); return ( <> <div className={classes.label}></div> </> ); }
