Ипользование

Установка

npm i --save @material-ui/core

Варианты импортов

import Button from '@material-ui/core/Button'; import { Button } from '@material-ui/core';

Простой пример

import { Button } from '@material-ui/core'; const Material = () => { return ( <Button variant="contained" color="primary"> Hello World </Button> ); } export default Material;

Стилизация

import { withStyles } from '@material-ui/core/styles'; import { Button } from '@material-ui/core'; import styles from './MaterialStyles'; const Material = ({ classes }) => { return ( <div className={classes.container}> <Button variant="contained" color="primary" onClick={() => { alert('clicked') }} classes={{ root: classes.root }} > Hello World </Button> </div> ); } export default withStyles(styles)(Material); const MaterialStyles = () => ({ container: { border: '2px solid blue', }, root: { color: 'yellow' } }); export default MaterialStyles;
