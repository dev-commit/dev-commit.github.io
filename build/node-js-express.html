Информация

Установка и настройка

npm i --save express

Структура

1. Подключение и вызов Express

const express = require('express'); const app = express();

2. Марштуризация

app.get('/url', (req, res) => { // Отображаение текста res.send('This is Text'); // Отображаение HTML res.send('<h1>This is HTML</h1>'); // Отображаение HTML файла res.sendFile(__dirname + '/index.html'); });

3. Прослушивание порта

const PORT = process.env.PORT || 3000; app.listen(PORT, '127.0.0.1');

Общее

const app = express(); // app.use(express.json()); // req - Readable, res - Writable app.post("path", parser, (req, res) => { // может получить req.body, только если указали app.use(express.json()) // иначе будет undefined, т.к. это stream console.log(req.body); }); // Можно добавить свой Middleware // parser - Middleware app.post("path", parser, (req, res) => { // можем получить req.body, только если указали app.use(express.json()) // иначе будет undefined, т.к. это stream console.log(req.body); });

Маршрутизация

1. Статическая ссылка

// главная app.get('/', (req, res) => { res.send('This is Index'); }); // news app.get('/news', (req, res) => { res.send('This is News'); });

2. Динамические параметры в URL

// > "/news/starter" app.get('/news/:name', (req, res) => { const params = req.params; // => { name: 'starter' } }); // > "/news/starter/23" app.get('/news/:name/:id', (req, res) => { const params = req.params; // => { name: 'starter', id: '23' } });

3. Query параметры из URL

// > "/news?id=23&city=Melbourne" app.get('/news', (req, res) => { const query = req.query; // => { id: '23', city: 'Melbourne' } });

get()

const express = require('express'); const app = express(); app.get('/url', (req, res) => { // });

use()

  • Express Middleware
const express = require('express'); const app = express(); app.use((req, res, next) => { res.status(201).end('Hello'); })

Информация

Установка и настройка

npm i --save ejs

Структура

› index.js 📂 public › styles.css 📂 views › news.ejs 📂 blocks › header.ejs

Пример

index.js

const express = require('express'); const app = express(); app.use('/public', express.static('public')); // Указание шаблонизатора app.set('view engine', 'ejs'); app.get('/news', (req, res) => { // Берем шаблон "news.ejs" из "views" // Передаем объект с ключами: "value" и "arr" res.render('news', { value: 'Value', arr: [1, 2, 3, 4] }); }); app.listen(3000);

views/news.ejs

  • В "blocks/header" обычный html-код
<%- include('blocks/header'); %> <%= value %> <ul> <% arr.map(item => { %> <li><%= item %></li> <% }); %> </ul>

Tags

Подключение файла "header"

<%- include('blocks/header'); %>

Вывод переменной "value"

<%= value %>

Вывод массива "arr"

<% arr.map(item => { %> <div><%= item %></div> <% }); %>

Статичные файлы

  • Нужно явно указать ссылку на статичные файлы: css, img, doc и т.д., иначе эти файлы не будут загружаться
  • Статичные файлы размещаются в директории public
› index.js 📂 public › styles.css 📂 views › about.ejs

1. Регистрация в index.js

const express = require('express'); const app = express(); app.use('/public', express.static('public'));

2. Задание стилей

h1 { color: yellowgreen; }

3. Добавление стилей на страницу

<link rel="stylesheet" href="/public/styles.css"> <h1>About</h1>
