Маршрутизация на Express

const express = require('express'); const app = express(); // статическая ссылка (главная) app.get('/', (req, res) => { res.send('This is home'); }); // статическая ссылка (news) app.get('/news', (req, res) => { res.send('This is news'); }); // ссылка с name статьи /news/23 app.get('/news/:name', (req, res) => { res.send('id=' + req.params.name); }); // ссылка с name и id статьи /news/23 app.get('/news/:name/:id', (req, res) => { res.send('id=' + req.params.name + req.params.id); }); // отображение html файла app.get('/', (req, res) => { res.sendFile(__dirname + '/index.html'); }); app.get('/about', (req, res) => { res.sendFile(__dirname + '/about.html'); }); app.listen(3000);

Установка шаблонизатора ejs

npm i ejs

index.js

const express = require('express'); const app = express(); // указание шаблонизатора, который будем использовать app.set('view engine', 'ejs'); // ссылка на статические файлы (css, img, doc и т.д. иначе не будут загружаться) app.use('/public', express.static('public')); app.get('/news/:id', (req, res) => { // берет шаблон news.ejs из папки views // передает свойство newsId = req.params.id // передает переменную newsParam = 234 // передаем объект newsParam const obj = {title: 'Новость', id: 4, paragraphs: ['Текст', 'Числа: 2, 4', 77]}; res.render('news', {newsId: req.params.id, newsParam: 234, newsObj: obj}); }); app.get('/about', (req, res) => { res.render('about'); }); app.listen(3000);
