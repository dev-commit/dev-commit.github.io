require, module.exports

Экспорт одного значения

Экспорт нескольких значений

Экспорт 1

const a = 'Hello'; const b = 'World'; module.exports.a = a; module.exports.b = b;

Экспорт 2

module.exports.a = 'Hello'; module.exports.b = 'World';

Экспорт 3

const a = 'Hello'; const b = 'World'; module.exports = { a: a, b: b, };

Испорт

const data = require('./data'); const a = data.a; const b = data.b;
