Модули ES6

    Особенности
  • Не блокируют парсинг html
  • Выполняются отложенно
  • Имею свою область видимости
  • Код модуля всегда выполняется в строгом режиме
  • Для экспорта обязательно нужно имя

import

// Импорт сущности, экспортированной по-умолчанию import MyClass from './modules/lib' // Импорт одной сущности import { libFunction } from './modules/lib' // Импорт нескольких сущностей import { libFunction, libConst } from './modules/lib' // Совмещение различных сущностей import MyClass, { libFunction } from './modules/lib' // Задание псевдонима для импортируемой сущности import { libConst as name } from './modules/lib' // Если не указывать файл, то import index.js из "ui" import './ui' // Импорт модуля как объекта (импорт всего, что экспортирует модуль) import * as obj from './modules/lib' // Импорт модуля как объекта, значениями свойств которого будут экспортированные значения (obj.libConst) import './modules/lib' as obj // Импорт всех данных (в "./modules/lib" нет "export"), запуск кода из модуля import './modules/lib'

export

// Экспорт сущности по-умолчанию export default MyClass // Экспорт одного значения export const a = 22; // Экспорт нескольких сущностей (const one, const two) export { one, two }; // Задание псевдонима при экспорте сущностей export { one as once, two as twice } // Экспорт функции export function libFunction() { console.log(libFunction); }
