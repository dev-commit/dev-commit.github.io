createBrowserRouter

Параметры "element" и "Component"

  • В параметрах может быть element или Component
import { createBrowserRouter } from "react-router/dom" const About = () => <h1>Hello world</h1> const router = createBrowserRouter([ { path: "/", element: <div>Hello World</div>, }, { path: "about", Component: About }, ])

Вложенные маршруты

  • Маршруты могут быть вложены в родительские маршруты с помощью children
  • Дочерние маршруты отображаются через <Outlet/> в родительском маршруте
export const router = createBrowserRouter([ { path: "/", Component: App, children: [ { index: true, Component: Home, }, { path: "about", Component: About, }, ], }, ]) import { Outlet } from "react-router" const App = () => { return ( <div> <Outlet /> </div> ) }
