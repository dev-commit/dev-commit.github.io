Pick и Record

  • Pick - выбрать указанные поля
  • Record -

Определение типов

type Pick<T, K extends keyof T> = { [P in K]: T[P]; } type Record<K extends string, T> = { [P in K]: T; }

Pick<>

type Dog = { id: number; name: string; weight: number; age: number; } // взяли только "id" и "name" type Doggy = Pick<Dog, 'id' | 'name' >; // эквивалентно // type Doggy = { // id: number; // name: string; // } const obj: Doggy = { id: 23, name: 'Jack' }

Record<>

type Dog = { id: number; name: string; } type Doggy = Record<'prop1' | 'prop2' | 'prop3', Dog>; // эквивалентно // type Doggy = { // prop1: Dog; // prop2: Dog; // prop3: Dog; // } const obj: Doggy = { prop1: { id: 23, name: 'Jack' }, prop2: { id: 23, name: 'Jack' }, prop3: { id: 23, name: 'Jack' } }
