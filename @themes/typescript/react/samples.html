Примеры типизации

...rest

  • Задача типизировать пропсы:
  • Есть компонент-обёртка
  • Непосредственно в компоненте нужен проп "label" - остальные пропы прокидывает дальше
  • Пробрасываемое количество пропов может быть любое
import React from 'react'; type Props = { label: string } & Parameters<typeof ComponentInput>[0]; const ComponentWrapper: React.FC<Props> = ({ label, ...rest }) => { return ( <> <span>{label}</span> <ComponentTarget {...rest} /> </> ); };
