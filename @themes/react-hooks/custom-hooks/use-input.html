useInput

import React, { useState } from 'react'; const useInput = (initialValue) => { const [value, setValue] = useState(initialValue); const onChange = event => setValue(event.target.value); const clear = () => setValue(''); return { bind: { value, onChange }, value, clear } }

Usage

import React from 'react'; const App = () => { const input = useInput(''); return ( <> <input type="text" {...input.bind} /> <button onClick={() => input.clear()}>Очистить</button> <span>{input.value}</span> </> ) }
