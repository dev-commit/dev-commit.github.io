useLogger

import React, { useState, useEffect } from 'react'; const useLogger = (value) => { useEffect(() => { console.log('Value changed: ', value); }, [value]); }

Usage

const App = () => { const [value, setValue] = useState(''); useLogger(value); const onChange = e => setValue(e.target.value); return ( <> <input type="text" value={value} onChange={onChange} /> <h1>{value}</h1> </> ) }
