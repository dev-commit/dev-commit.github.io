Добавить свойство с проверкой

  • Добавлять свойство только в том случае, если оно передано

Основной пример

const age = 23 const obj = { name: 'Tony', ...(age && { age }) }

Пример с флагом

const flag = true; const obj = { name: 'Tony', ...(flag && { age: 23 }) } console.log(obj); const flag = true; const age = flag && { age: 23 } const obj = { name: 'Tony', ...age } // => { name: 'Tony', age: 23 } console.log(obj);

React Component

const flag = true; <App {...(flag && { title: 'Hello' })} />
