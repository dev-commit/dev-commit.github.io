Custom `Document`

https://nextjs.org/docs/advanced-features/custom-document
  • Обычно используется для дополнения тегов <html> и <body> приложения
  • Работает в рамках всего приложения
  • Файл: ./pages/_document.js
./pages/_document.js
  • В примере, добавляется ссылка на Google Fonts для всего приложения
import Document, { Html, Head, Main, NextScript } from 'next/document'; class MyDocument extends Document { render() { return ( <Html> <Head> <link href="https://fonts.googleapis.com/css2?family=Bakbak+One&display=swap" rel="stylesheet" /> </Head> <body> <Main /> <NextScript /> </body> </Html> ) } } export default MyDocument;
