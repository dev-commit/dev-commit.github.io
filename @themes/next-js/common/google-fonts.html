Добавление Google Fonts

Добавление ссылки с Google Fonts

./pages/_document.js import Document, { Html, Head } from 'next/document'; const App = () => { return ( <Html> <Head> <link href="https://fonts.googleapis.com/css2?family=Bakbak+One&display=swap" rel="stylesheet" /> </Head> </Html> ) } export default App;

Использование шрифтов

./pages/_app.js // Подключение глобального CSS import '../styles/main.css'; ./styles/main.css body { font-family: 'Bakbak One', cursive; }
