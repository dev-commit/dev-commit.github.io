Общие методы

force

  • Если элемент находится под другими элементами
cy.get('div').click({ force: true }); cy.get('div', { force: true }).blur();

find()

  • Получить дочерние элементы DOM определенного селектора
// получит все элементы <tr> cy.get('table').find('tbody')

eq

  • Получить элемент DOM по определенному индексу в массиве элементов
// Первый <tr> таблицы должен содержать "Tony" cy.get('table tbody tr').eq(0).contains('Tony')

first()

// Yield first link in nav cy.get('nav a').first()

cy.go()

cy.go('back')

cy.contains()

  • Найти контент на странице по тексту
cy.contains('button', 'Emergency Contact').click()

cy.get().within

  • Охватывает все последующие команды cy внутри этого элемента. Полезно при работе с определенной группой элементов, например <form>
cy.get('.container').within(() => { cy.get('.item').should('contain', 'Hello'); }); cy.location('search').should('contain', 'query=fish'); cy.get('.chip').children('.chip__btn').click(); cy.log('Choose tag');
