Условие равенства

Постановка задачи

// Необходимо чтобы было "true" console.log(a == true && a == false);

Решение 1

let c = 0; const a = Object.create({ [Symbol.toPrimitive]: () => { if (c === 0) { c++; return true; } return false; }});

Решение 2

let a = { toggler: false, } a.valueOf = function() { this.toggler = !this.toggler; return this.toggler } console.log(a == true && a == false)
