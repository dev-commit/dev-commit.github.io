Палиндром

  • Палиндром - число, буквосочетание, слово или текст, одинаково читающееся в обоих направлениях
  • for вероятно можно заменить на every
const isPalindrom = str => { // Для предложений нужны дополнительные действия const resultStr = str.toLowerCase().split(' ').join(''); // Находим середину слова const length = Math.round(resultStr.length / 2); for (let i = 0; i < length; i++) { if (resultStr[i] !== resultStr[resultStr.length - i - 1]) { return false; } } return true; } console.log(isPalindrom('ротор')); // true console.log(isPalindrom('ротор1')); // false console.log(isPalindrom('дом')); // false console.log(isPalindrom('А луна канула')); // true
