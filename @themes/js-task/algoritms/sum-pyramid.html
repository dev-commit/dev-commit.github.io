Простое решение

const foo = number => { let counter = 0; let sum = 0; for (let i = 1; i <= number; i++) { for (let j = 1; j <= i; j++) { counter++; } } for (let i = 1; i <= counter; i++) { sum += i; } return sum; } const a1 = foo(1); // => 1 const a2 = foo(2); // => 6 const a3 = foo(3); // => 21 const a4 = foo(4); // => 55 const a5 = foo(5); // => 120 const a6 = foo(6); // => 231
