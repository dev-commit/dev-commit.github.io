MergeSort

const mergeSort = (arr1, arr2) => { const result= []; let l1 = arr1.legth; let l2 = arr2.length; while(result.length !== l1+l2) { if(arr1[0] < arr2[0]) { result.push(arr1.shift()); } else { result.push(arr2.shift()); } } return result; } mergeSort([1,3,5], [2,4,6]) // [1, 2, 3], [4, 5, 6] // => [1, 2, 3, 4, 5, 6]
