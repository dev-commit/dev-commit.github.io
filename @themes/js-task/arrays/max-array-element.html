const arr = [1, 2, 3, 77, 4]; const max = Math.max.apply(null, arr); // 77 (контекстом будет объект Math) const max = Math.max.apply(Math, arr); // 77 (контекстом будет объект Math) - аналогично // в данном случае в качестве контекста можно передавать что угодно, // поскольку в своей внутренней реализации метод Math.max не использует this const numbers = [1,2,3,4]; Math.max(...numbers);
