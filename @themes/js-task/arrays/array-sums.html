Двумерный массив сумм

const arr = [1,2,3,5,6,7,0,-1] const sum = 6 function doit(arr, sum) { const obj = {}; let res = []; arr.forEach(item => { if(obj[item]) { obj[item]++; } else { obj[item] = 1; } }); arr.forEach(item => { obj[item]--; if (obj[sum-item]) { res.push([item, sum-item]); } }); return res; } console.log(doit(arr, sum)) // [[1,5], [6,0], [7, -1]]
