Превращение массива неогранчиенной вложенности в плоский

Рещение 1

const arr = [1,2,3,[1,2,3,4, [2,3,4]]]; function flattenDeep(arr) { let result = []; arr.forEach(element => { if (typeof element !== 'object') { result.push(element); } else { result.push(...flattenDeep(element)); } }) return result; } console.log(flattenDeep(arr)); // [1, 2, 3, 1, 2, 3, 4, 2, 3, 4] flattenDeep(arr); // undefined

const deepArray = [1, 2, [3, 4, [5, 6, 7]]]; const flatten = deepArray => { const ans = []; const innerFn = el => { if (!Array.isArray(el)) { ans.push(el); } else { innerFn(el); } } deepArray.forEach(innerFn); return ans; } flatten(deepArray); // => [1, 2, 3, 4, 5, 6, 7]
