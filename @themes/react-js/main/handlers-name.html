Название обработчиков событий

  • on - функция внутри компонента. Может приходить из props
  • handle / handler - функции внутри компонента, которые передаются в props другому компоненту
const App = () => { // Используется внутри компонента const onChange = () => null // Передается в props другого компонента const handleClick = () => null return ( <> <input onChange={onChange} /> <Child onClick={handleClick} /> </> ) } const Child = (props) => { // Берется из props const { onClick } = props return null }

  • onClick onButtonClick
  • onChange onInputChange
