Math

Округление

const a1 = Math.round(5.2); // => 5, округляет число к ближайшему целому (5 округляется вверх) const a2 = Math.ceil(5.1); // => 6, округляет число в большую сторону const a3 = Math.floor(5.9); // => 5, округляет число в меньшую сторону

Random

// Генерация случайного дробного числа от 0 (включительно) до 1 (не включая) // н-р: 0.27647450465282075 const a = Math.random();
Варианты
const a1 = Math.floor(Math.random() * 10); // 0 - 9 const a2 = Math.floor(Math.random() * 6); // 0 - 5 const a3 = Math.floor(Math.random() * 2); // 0 - 1 const a4 = Math.floor(Math.random() * 101); // 0 - 100

Методы

const a1 = Math.pow(4,2); // степень 4 в 2 const a2 = Math.sqrt(400); // квадратный корень числа const a3 = Math.abs(-7); // модуль const a4 = Math.max(1, 2, 3, 4); // большее число из аргументов const a5 = Math.min(1, 2, 3, 4); // меньшее число из аргументов const a6 = Math.exp(2); // возведение часла "e" в указанную степень const a7 = Math.log(5); // натуральный логарифм числа const a1 = Math.PI; // число PI const a2 = Math.E; // чисто E const a1 = Math.sin(1); // синус (радиан) const a2 = Math.cos(1); // косинус const a3 = Math.tan(1); // тангенс const a4 = Math.atan(1); // арктангенс const a5 = Math.asin(1); // арксинус const a6 = Math.acos(1); // арккосинус
