Форматирование строк. Метод format

# если для подстановки требуется только один аргумент, то значение - сам аргумент a = 'Hello {}'.format('Tony') # 'Hello, Tony!' a = 'Hello {name}'.format(name='Tony') # 'Hello, Tony!' # если для подстановки требуется несколько аргументов a = '{0}, {1}, {2}'.format('a', 'b', 'c') # 'a, b, c'
