useSWRMutation

Хук для удаленных мутаций. Удаленные мутации запускаются только вручную, а не автоматически, как useSWR URL для запроса Асинхронная функция, в которой происходит запрос. принимает URL, переданный первым аргументом в useSWRMutation Необязательный объект со свойствами { data, error, isMutating, trigger, reset }
Return
  • data - данные для заданного ключа, возвращаемые из fetcher
  • error - ошибка, выданная fetcher (или undefined)
  • isMutating - если есть текущая удаленная мутация
  • trigger(arg, options) - функция для запуска удаленной мутаци (выполнение запроса к API)
  • reset() - функция для сброса состояния (data, error, isMutating)

Пример

export const useSampleMutation = () => { const { data, error, isMutating, trigger, reset } = useSWRMutation( "https://jsonplaceholder.typicode.com/todos/1", async (url) => { const res = await fetch(url) return await res.json() }, ) return { data, error, isMutating, trigger, reset, } } const Profile: React.FC = () => { const { trigger, isMutating, error } = useSampleMutation() const onClick = async () => { try { const result = await trigger() console.log(result) } catch (e) { // Обработка ошибки console.log(e) } } // Возможно, не нужно использовать try/catch в onClick // В error отображается аналогичная ошибка console.log("error", error) return ( <button disabled={isMutating} onClick={onClick}> Click </button> ) }
