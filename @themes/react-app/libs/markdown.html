import { useState, useEffect } from "react" import Markdown from "react-markdown" const App = () => { const [text, setText] = useState("") useEffect(() => { import(`./text.md`) .then((res) => { fetch(res.default) .then((res) => res.text()) .then((res) => setText(res)) .catch((err) => console.log(err)) }) .catch((err) => console.log(err)) }) return ( <div> <Markdown>{text}</Markdown> </div> ) }
