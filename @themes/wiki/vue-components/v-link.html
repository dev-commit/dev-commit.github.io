v-link

  • url (string) - Отображение URL
  • title (string) - Отображение title вместо URL
  • block (boolean) - Ссылка становится блочным элементом
  • text (string) - Дополнительный текст рядом с URL
<v-link url="https://yandex.ru/" />
<v-link url="https://yandex.ru/" title="Link Title" />
<v-link url="https://yandex.ru/" block="true" />
<v-link url="https://yandex.ru/" text="Yandex" />
<v-link url="https://yandex.ru/" title="Link Title" text="Yandex" />
