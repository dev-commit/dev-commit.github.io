<input type="checkbox"> (галочка)

const checkbox = $('#checkbox').prop('checked'); // читаем (true/false) $('#checkbox').prop('checked', true); // устанавливаем значение checked (true/false)

<input type="radio"> (переключатель)

const radio = $('input[name=radio]:checked').prop('id'); // получаем id выбранного radio const radio = $('#radio1').prop('checked'); // проверяем по id выбран ли radio (true/false) $('#radio1').prop('checked', true); // устанавливаем checked по id

<select></select> (выпадающий список)

const value = $('#select').val(); // читаем выбранный value const text = $('#select option:selected').text(); // читаем выбранный text const text = $('#select [value=2]').text(); // читаем text по заданному value $('#select [value=3]').attr('selected', 'selected'); // устанавливаем заданный элемент в selected
