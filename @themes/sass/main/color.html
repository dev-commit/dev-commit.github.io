Операции над цветами

lighten($color,10%) # осветляет исходный цвет darken($color,10%) # затемняет исходный цвет saturate($color,70%) # увеличивает насыщенность цвета от исходного desaturate($color,70%) # уменьшает насыщенность цвета от исходного transparentize($color,0.2) # добавляет к исходному цвету альфа-канал opacify($color, 0.1) # задание менее прозрачного цвета transparentize($color, 0.4) # задание более прозрачного цвета rgba(#333, 0.75) # перевод из hex в rgba
