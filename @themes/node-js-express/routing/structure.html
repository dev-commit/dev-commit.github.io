Структура

1. Подключение и вызов Express

const express = require('express'); const app = express();

2. Марштуризация

app.get('/url', (req, res) => { // Отображаение текста res.send('This is Text'); // Отображаение HTML res.send('<h1>This is HTML</h1>'); // Отображаение HTML файла res.sendFile(__dirname + '/index.html'); });

3. Прослушивание порта

const PORT = process.env.PORT || 3000; app.listen(PORT, '127.0.0.1');
