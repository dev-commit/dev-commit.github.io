Пример

import { useFormikContext } from 'formik'; const ComponentWithContext = ({ name }) => { const { values, errors, initialValues, setValues, setFieldValue } = useFormikContext(); // console.log(values, errors); const changeValue = () => { setFieldValue(name, "Hello"); }; // Записать во все values дефолтные значения const defaultValues = () => { setValues(initialValues); }; return ( <> <button type="button" onClick={changeValue}>Change Value</button> <button type="button" onClick={defaultValues}>Default Values</button> </> ); } import { useField } from 'formik'; const Input = ({ name, type }) => { const [field, , { setValue }] = useField(name); return ( <input type={type} name={field.name} value={field.value} onChange={e => setValue(e.target.value)} /> ); }; import { Form, Formik } from 'formik'; const App = () => { const initialFormData = { userName: "", }; const handleSubmit = (values) => { console.log(values); }; return ( <Formik initialValues={initialFormData} onSubmit={handleSubmit}> <Form> <Input name="userName" type="text" /> <ComponentWithContext name="userName" /> <button type="submit">Submit</button> </Form> </Formik> ); }
