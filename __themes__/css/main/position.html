Position

div { position: static; # статичное (стандартное позиционирование) position: relative; # относительное (положение устанавливается относительно его исходного места) position: absolute; # абсолютное (отсчёт координат ведётся от края окна браузера, если у родителя не установлено position: relative) position: fixed; # фиксированное (привязывается к указанной свойствами left, top, right и bottom точке на экране) position: sticky; # сочетание относительного и фиксированного позиционирования }
