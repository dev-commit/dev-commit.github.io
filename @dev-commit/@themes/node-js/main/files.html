Файлы и директории

Библиотека fs

const fs = require('fs');

Чтение из файла

// синхронно const fileSync = fs.readFileSync('text.txt', 'utf8'); // асинхронно const fileASync = fs.readFile('text.txt', 'utf8'); // асинхронно fs.readFile('text.txt', 'utf8', function(err, data) { console.log(data); });

Запись в файл

// синхронно fs.writeFileSync('text.txt', 'Hello \n World'); // асинхронно fs.writeFile('text.txt', 'Hello \n World', function(err, data) {});

Удаление файла

// синхронно fs.unlinkSync('file.txt'); // асинхронно fs.unlink('file.txt', function() {});

Проверка существования файла или директории

fs.existsSync('stuffs');

Создание директории

// синхронно fs.mkdirSync('stuff'); // асинхронно fs.mkdir('stuff');
  • При использовании асинхронного метода, необходимо использовать функцию обратного вызова
fs.mkdir('stuff', function() { fs.readFile('readme.txt', 'utf8', function(err, data) { fs.writeFile('./stuff/writeme.txt', data); }); });

Удаление пустой директории

// синхронно fs.rmdirSync('stuff'); // асинхронно fs.rmdir('stuff');

Удаление непустой директории

const rimraf = require('rimraf'); // удаление не пустой директории fs.mkdirSync('nodejs/pages'); // удаление сначала файла из папки fs.unlink('./test-folder/file.txt', function() { // после этого удаление папки fs.rmdir('test-folder', function() {}); });
