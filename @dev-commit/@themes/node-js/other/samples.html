Exports функции 2

greet.js module.exports = function(greeting) { return function(name) { return `${greeting}, ${name}`; } } index.js const getGreet = require('./greet'); const greet = getGreet('Привет'); console.log(greet('Tony')); console.log(greet('Harvey')); console.log(greet('Andy')); Результат // => Привет, Tony // => Привет, Harvey // => Привет, Andy
